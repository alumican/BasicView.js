 10 /**
 11  * BasicView class is very simple template for getting started Three.js.
 12  * @class
 13  * @param {string} [rendererType="WebGL"] Renderer type ("WebGL" or "Canvas" or "SVG" or "DOM").
 14  * @param {object} [rendererParams={}] Renderer construct parameters.
 15  * @param {string} [cameraType="Perspective"] Camera type ("Perspective" or "Orthographic").
 16  * @param {bool} [useStats=true] Whether stats is used or not.
 17  * @param {bool} [useAutoScalingToStage=true] Whether content is scaled to fit stage (true) or not.
 18  * @param {integer} [viewportWidth=640] Viewport width. If auto scaling is used, this value is ignored.
 19  * @param {integer} [viewportHeight=480] Viewport height. If auto scaling is used, this value is ignored.
 20  *
 21  * @property {DOM element} container Viewport Container.
 22  * @property {Three.js Scene object} scene Three.js Scene Object.
 23  * @property {Three.js Camera object} camera Three.js Camera Object.
 24  * @property {Three.js Renderer object} renderer Three.js Renderer Object.
 25  * @property {integer} mouseX Current mouse position x.
 26  * @property {integer} mouseY Current mouse position y.
 27  * @property {integer} oldMouseX Previous mouse position x.
 28  * @property {integer} oldMouseY Previous mouse position y.
 29  * @property {integer} screenWidth Window width.
 30  * @property {integer} screenHeight Window height.
 31  * @property {bool} isMouseDown Whether mouse is being pressed.
 32  * @property {bool} isMouseDragging Whether mouse is being dragged.
 33  * @property {bool} isKeyDown Whether keyboard is being pressed.
 34  * @property {string} rendererType Renderer Type.
 35  * @property {string} cameraType Camera Type.
 36  * @property {bool} useAutoScalingToStage Whether content is scaled to fit stage (true) or not.
 37  *
 38  * @return void
 39  */
 40 function BasicView(rendererType, rendererParams, cameraType, useStats, useAutoScalingToStage, viewportWidth, viewportHeight)
 41 {
 42 	this.container;
 43 
 44 	this.scene;
 45 	this.camera;
 46 	this.renderer;
 47 
 48 	this.mouseX;
 49 	this.mouseY;
 50 	this.oldMouseX;
 51 	this.oldMouseY;
 52 
 53 	this.screenWidth;
 54 	this.screenHeight;
 55 
 56 	this.isMouseDown;
 57 	this.isMouseDragging;
 58 	this.isKeyDown;
 59 	
 60 	this.rendererType;
 61 	this.cameraType;
 62 	
 63 	this.useAutoScalingToStage;
 64 
 65 	//----------------------------------------
 66 	//call construntor
 67 	this._initialize.apply(this, arguments);
 68 }
 69

104 	{

108 	 * This function is called when window is closing.
109 	 * @event
110 	 * @return void
111 	 */
112 	exit : function()
113 	{
114 	},

117 	 * This function is called when mouse button is pressed.
118 	 * @event

120 	 * @return void
121 	 */
122 	onMouseDown : function(event)
123 	{
124 	},

127 	 * This function is called when mouse button is released.

189 	/**
190 	 * #@+
191 	 * @private
192 	 */
193 	_initialize : function(rendererType, rendererParams, cameraType, useStats, useAutoScalingToStage, viewportWidth, viewportHeight)
194 	{
195 		//set default value
196 		if (rendererType == undefined) rendererType = "WebGL";
197 		if (rendererParams == undefined) rendererParams = {};
198 		if (cameraType == undefined) cameraType = "Perspective";
199 		if (useStats == undefined) useStats = true;
200 		if (useAutoScalingToStage == undefined) useAutoScalingToStage = true;
201 		if (viewportWidth == undefined) viewportWidth = 640;
202 		if (viewportHeight == undefined) viewportHeight = 480;
203 
204 		//output version
205 		if (useStats)
206 		{
207 			console.log("BasicView " + BasicView.version);
208 		}
209 
210 		//WebGL is not supported
211 		if(rendererType == "WebGL" && !Detector.webgl)
212 		{
213 			Detector.addGetWebGLMessage();
214 			return;
215 		}
216 
217 		//setup
218 		{
219 			this.rendererType = rendererType;
220 			this.cameraType = cameraType;
221 			this.useAutoScalingToStage = useAutoScalingToStage;
222 
223 			this.isMouseDown = false;
224 			this.isMouseDragging = false;
225 			this.isKeyDown = false;
226 
227 			this.oldMouseX = this.mouseX = 0;
228 			this.oldMouseY = this.mouseY = 0;
229 
230 			var self = this;
231 			$(function() {
232 
233 				if (self.useAutoScalingToStage)
234 				{
235 					self.screenWidth  = $(window).width();
236 					self.screenHeight = $(window).height();
237 				}
238 				else
239 				{
240 					self.screenWidth  = viewportWidth;
241 					self.screenHeight = viewportHeight;
242 				}
243 
244 				//create scene
245 				self.scene = new THREE.Scene();
246 
247 				//create camera
248 				switch (cameraType)
249 				{
250 					case "Orthographic":
251 						self.camera = new THREE.OrthographicCamera(-self.screenWidth * 0.5, self.screenWidth * 0.5, self.screenHeight * 0.5, -self.screenHeight * 0.5, 1, 10000);
252 						break;
253 
254 					default:
255 						self.cameraType = "Perspective";
256 						self.camera = new THREE.PerspectiveCamera(80, self.screenWidth / self.screenHeight, 1, 10000);
257 						break;
258 				}
259 				self.camera.position.z = 200;
260 
261 				//create renderer
262 				switch (rendererType)
263 				{
264 					case "Canvas":
265 						self.renderer = new THREE.CanvasRenderer(rendererParams);
266 						break;
267 
268 					case "SVG":
269 						self.renderer = new THREE.SVGRenderer();
270 						break;
271 
272 					case "DOM":
273 						self.renderer = new THREE.DOMRenderer();
274 						break;
275 
276 					default:
277 						this.rendererType = "WebGL";
278 						self.renderer = new THREE.WebGLRenderer(rendererParams);
279 						break;
280 				}
281 				self.renderer.setSize(self.screenWidth, self.screenHeight);
282 
283 				//create container
284 				self.container = $("<div>").get(0);
285 				$(document.body).append(self.container);
286 
287 				//stats
288 				if (useStats)
289 				{
290 					self.stats = new Stats();
291 					$(self.stats.domElement).css("position", "absolute");
292 					$(self.stats.domElement).css("top", "0px");
293 					$(self.container).append(self.stats.domElement);
294 				}
295 
296 				//container
297 				self.container.appendChild(self.renderer.domElement);
298 				self.scene.add(self.camera);
299 
300 				//event
301 				$(document).mousedown(function(e) { self._mouseDownHandler(e); });
302 				$(document).mousemove(function(e) { self._mouseMoveHandler(e); });
303 				$(document).mouseup(function(e) { self._mouseUpHandler(e); });
304 				$(window).keydown(function(e){ self._keyDownHandler(e); });
305 				$(window).keyup(function(e){ self._keyUpHandler(e); });
306 				$(window).resize(function(e) { self._resizeHandler(e); });
307 				$(window).unload(function(e) { self._unloadHandler(e); });
308 
309 				self.setup();
310 				self._enterFrameHandler(null);
311 			});
312 		}
313 	},
314 
315 	_enterFrameHandler : function(e)
316 	{
317 		var self = this;
318 		requestAnimationFrame(function(e) { self._enterFrameHandler(e); });
319 
320 		this.oldMouseX = this.mouseX;
321 		this.oldMouseY = this.mouseY;
322 
323 		this.update();
324 		this.draw();
325 
326 		this.stats.update();
327 	},
328 
329 	_mouseDownHandler : function(e)
330 	{
331 		this.isMouseDown = true;
332 		this.onMouseDown(e);
333 	},
334 
335 	_mouseMoveHandler : function(e)
336 	{
337 		if (e.pageX == this.mouseX && e.pageY == this.mouseY) return;
338 		this.mouseX = e.pageX;
339 		this.mouseY = e.pageY;
340 
341 		if (this.isMouseDown)
342 		{
343 			this.isMouseDragging = true;
344 			this.onMouseDragged(e);
345 		}
346 		else
347 		{
348 			this.onMouseMove(e);
349 		}
350 	},
351 
352 	_mouseUpHandler : function(e)
353 	{
354 		this.isMouseDown = false;
355 		this.isMouseDragging = false;
356 		this.onMouseUp(e);
357 	},
358 
359 	_keyDownHandler : function(e)
360 	{
361 		this.isKeyDown = true;
362 		this.onKeyDown(e.keyCode, e);
363 	},
364 
365 	_keyUpHandler : function(e)
366 	{
367 		this.isKeyDown = false;
368 		this.onKeyUp(e.keyCode, e);
369 	},
370 
371 	_resizeHandler : function(e)
372 	{
373 		if (this.useAutoScalingToStage)
374 		{
375 			this.screenWidth  = $(window).width();
376 			this.screenHeight = $(window).height();
377 		}
378 
379 		if (this.useAutoScalingToStage)
380 		{
381 			switch (this.cameraType)
382 			{
383 				case "Orthographic":
384 					this.camera.aspect = this.screenWidth / this.screenHeight;
385 					this.camera.left = -this.screenWidth * 0.5;
386 					this.camera.right = this.screenWidth * 0.5;
387 					this.camera.top = this.screenHeight * 0.5;
388 					this.camera.bottom = -this.screenHeight * 0.5;
389 					this.camera.updateProjectionMatrix();
390 					this.renderer.setSize(this.screenWidth, this.screenHeight);
391 					break;
392 
393 				default:
394 					this.camera.aspect = this.screenWidth / this.screenHeight;
395 					this.camera.updateProjectionMatrix();
396 					this.renderer.setSize(this.screenWidth, this.screenHeight);
397 					break;
398 			}
399 		}
400 
401 		this.onResize(e);
402 	},
403 	
404 	_unloadHandler : function(e)
405 	{
406 		this.exit();
407 	}
408 	
409 	/** #@- */
410 }
